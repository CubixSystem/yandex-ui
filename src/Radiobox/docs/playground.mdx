import React, { useState } from 'react';
import { Radiobox } from '../desktop/bundle'

export const RadioboxStateful = (props) => {
    const [value, setValue] = useState(false);
    return (
        <Radiobox {...props} options={[
            {
                "label": "Option A",
                "value": "a"
            },
            {
                "label": "Option B",
                "value": "b"
            },
            {
                "label": "Option C (disabled)",
                "value": "c",
                "disabled": true
            }
        ]} onChange={() => setValue(!value)} />
    );
};

<Sandbox
    displayName='Radiobox'
    component={RadioboxStateful}
    knobs={{
        disabled: false,
        size: [['s', 'm'], 'm'],
        view: [['default', 'outline', ''], 'default']
    }}
/>
