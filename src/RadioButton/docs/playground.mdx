import React, { useState } from 'react';
import { RadioButton } from '../desktop/bundle'

export const RadioButtonStateful = (props) => {
    const [value, setValue] = useState(false);
    return (
        <RadioButton {...props} value={value} options={[
            { value: 'value1', children: 'Радио 1' },
            { value: 'value2', children: <>Радио 2</> },
            { value: 'value3', children: <><span>Радио</span> 3</> }
        ]} onChange={(event) => setValue(event.target.value)} />
    );
};

<Sandbox
    displayName='RadioButton'
    component={RadioButtonStateful}
    knobs={{
        disabled: false,
        size: [['s', 'm'], 'm'],
        view: [['default', ''], 'default']
    }}
/>
