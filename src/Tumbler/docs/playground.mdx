import { Tumbler as TumblerBase } from '../Tumbler.bundle/desktop'
import React, { useState } from 'react'

export const Tumbler = (props) => {
    const [checked, setChecked] = useState(false);
    return (
        <TumblerBase 
            {...props} 
            checked={checked} 
            onChange={() => setChecked(!checked)}
        />
    )
}
