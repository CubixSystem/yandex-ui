import React, { useState } from 'react';
import { Checkbox } from '../desktop/bundle'

export const CheckboxStateful = (props) => {
    const [value, setValue] = useState(false);
    return <Checkbox {...props} checked={value} onChange={() => setValue(!value)} />;
};

<Sandbox
    displayName='Checkbox'
    component={CheckboxStateful}
    knobs={{
        disabled: false,
        label: 'Label',
        size: [['s', 'm'], 'm'],
        view: [['default', ''], 'default'],
        indeterminate: false,
    }}
/>
