### Базовое использование

```ts
// src/App.ts
import React, { useState } from 'react'
import { useUpdateEffect } from '@yandex-lego/components/useUpdateEffect'

const App = () => {
  const [value, setValue] = useState('a')

  useUpdateEffect(() => {
      alert(value)
  }, [value])

  return (
      <> {value} </>
  )
}
```
